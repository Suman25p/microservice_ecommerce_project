$body = '{
  "productId": 1,
  "quantity": 2
}'

1..10 | ForEach-Object {
    try {
        $r = Invoke-WebRequest `
            -Uri "http://localhost:8089/orders" `
            -Method POST `
            -ContentType "application/json" `
            -Body $body `
            -UseBasicParsing

        Write-Host "SUCCESS $($r.StatusCode)"
    }
    catch {
        Write-Host "FAILED $($_.Exception.Response.StatusCode.value__)"
    }
}
